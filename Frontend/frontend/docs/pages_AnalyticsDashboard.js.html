/**
 * @file AnalyticsDashboard.js
 * @module AnalyticsDashboard
 * @description
 * Displays analytical visualizations for different data domains (Customers, Companies, Stocks).
 * Users can select which subject to analyze via a dropdown. Each subject shows corresponding charts.
 * The dashboard fetches aggregated analytics data from backend endpoints using a shared API client.
 *
 * @requires react
 * @requires recharts
 * @requires ../ApiRequest
 * @requires ./AnalyticsDashboard.css
 */

import { useEffect, useState } from 'react';
import { api } from '../ApiRequest';
import {
  LineChart, Line, XAxis, YAxis, Tooltip, CartesianGrid,
  BarChart, Bar, Legend, PieChart, Pie, Cell, ResponsiveContainer,
} from 'recharts';
import './AnalyticsDashboard.css';

/**
 * @function AnalyticsDashboard
 * @description
 * React component displaying analytics for Customers, Companies, and Stocks.
 * A dropdown lets users toggle which subjectâ€™s charts are visible.
 * Data is fetched once and stored in state to avoid redundant requests.
 *
 * @returns {JSX.Element} The rendered analytics dashboard component.
 */
export default function AnalyticsDashboard() {
  /** @member {string} selectedSubject - Currently selected analytics category. */
  const [selectedSubject, setSelectedSubject] = useState('customers');

  /** @member {Array<Object>} avgTenure - Average tenure data for customer segments. */
  const [avgTenure, setAvgTenure] = useState([]);

  /** @member {Array<Object>} monthlyJoins - Monthly customer acquisition trend data. */
  const [monthlyJoins, setMonthlyJoins] = useState([]);

  /** @member {Array<Object>} currencyDist - Company currency distribution data. */
  const [currencyDist, setCurrencyDist] = useState([]);

  /** @member {Array<Object>} volatility - Stock price volatility data. */
  const [volatility, setVolatility] = useState([]);

  /** @member {Array<string>} COLORS - Chart color palette for categorical visuals. */
  const COLORS = ['#4A6EF5', '#F57C00', '#00BFA5', '#BA68C8', '#43A047'];

  /**
   * @function useEffect
   * @description
   * Fetches analytics data from backend endpoints for all subjects (customers, companies, stocks).
   * Each dataset is normalized and coerced to numeric form for Recharts compatibility.
   * Executed once when the component mounts.
   */
  useEffect(() => {
    Promise.all([
      api.get('/customers/avg-tenure'),
      api.get('/customers/monthly-joins'),
      api.get('/companies/currency-distribution'),
      api.get('/stocks/volatility'),
    ])
      .then(([tenure, joins, currency, vol]) => {
        // --- Customers ---
        const fixedTenure = (tenure.rows || tenure).map((t) => ({
          ...t,
          avg_tenure: Number(t.avg_tenure),
        }));
        const fixedJoins = (joins.rows || joins).map((j) => ({
          ...j,
          new_customers: Number(j.new_customers),
        }));
        setAvgTenure(fixedTenure);
        setMonthlyJoins(fixedJoins);

        // --- Companies ---
        const fixedCurrency = (currency.rows || currency)
          .filter((c) => c.currency && c.company_count)
          .map((c) => ({
            currency: String(c.currency).trim(),
            company_count: Number(c.company_count) || 0,
          }));
        setCurrencyDist(fixedCurrency);

        // --- Stocks ---
        const fixedVol = (vol.rows || vol).map((v) => ({
          ...v,
          price_range: Number(v.price_range),
        }));
        setVolatility(fixedVol);
      })
      .catch((err) => console.error('Error fetching analytics:', err));
  }, []);

  return (
    <div className="analytics-container">
      <h1>Analytics Dashboard</h1>

      {/* ========== DROPDOWN SELECTOR ========== */}
      <div className="dropdown-container">
        <label htmlFor="subject-select">Select Subject:</label>
        <select
          id="subject-select"
          value={selectedSubject}
          onChange={(e) => setSelectedSubject(e.target.value)}
        >
          <option value="customers">Customers</option>
          <option value="companies">Companies</option>
          <option value="stocks">Stocks</option>
        </select>
      </div>

      {/* ========== CUSTOMERS SECTION ========== */}
      {selectedSubject === 'customers' && (
        <section>
          <h2>Customers</h2>

          <h3>Average Tenure by Segment</h3>
          <ResponsiveContainer width="100%" height={300}>
            <BarChart data={avgTenure}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis dataKey="segment" />
              <YAxis />
              <Tooltip />
              <Legend />
              <Bar dataKey="avg_tenure" fill="#4A6EF5" />
            </BarChart>
          </ResponsiveContainer>

          <h3>Monthly New Customer Joins</h3>
          <ResponsiveContainer width="100%" height={300}>
            <LineChart data={monthlyJoins}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis dataKey="month" />
              <YAxis />
              <Tooltip />
              <Legend />
              <Line type="monotone" dataKey="new_customers" stroke="#F57C00" />
            </LineChart>
          </ResponsiveContainer>
        </section>
      )}

      {/* ========== COMPANIES SECTION ========== */}
      {selectedSubject === 'companies' && (
        <section>
          <h2>Companies</h2>

          {currencyDist.length === 0 ? (
            <p>Loading currency distribution...</p>
          ) : (
            <ResponsiveContainer width="100%" height={350}>
              <PieChart>
                <Pie
                  data={currencyDist}
                  cx="50%"
                  cy="50%"
                  outerRadius={130}
                  dataKey="company_count"
                  nameKey="currency"
                  label={({ currency, company_count }) =>
                    `${currency}: ${company_count}`
                  }
                >
                  {currencyDist.map((entry, i) => (
                    <Cell key={`cell-${i}`} fill={COLORS[i % COLORS.length]} />
                  ))}
                </Pie>
                <Tooltip
                  formatter={(value, name, props) => [
                    value,
                    props.payload.currency || name,
                  ]}
                />
                <Legend />
              </PieChart>
            </ResponsiveContainer>
          )}
        </section>
      )}

      {/* ========== STOCKS SECTION ========== */}
      {selectedSubject === 'stocks' && (
        <section>
          <h2>Stock Volatility</h2>
          <ResponsiveContainer width="100%" height={300}>
            <BarChart data={volatility}>
              <CartesianGrid strokeDasharray="3 3" />
              <XAxis dataKey="ticker" />
              <YAxis />
              <Tooltip />
              <Legend />
              <Bar dataKey="price_range" fill="#BA68C8" />
            </BarChart>
          </ResponsiveContainer>
        </section>
      )}
    </div>
  );
}
