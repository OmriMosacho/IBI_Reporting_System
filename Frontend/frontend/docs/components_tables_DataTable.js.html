/**
 * @file DataTable.js
 * @module DataTable
 * @description
 * A reusable, fully-documented React component that dynamically fetches
 * and displays tabular data from the backend using the ApiRequest module.
 * It supports automatic row counting, collapsible view, loading and error states.
 */

import { useState, useEffect } from 'react';
import { api } from '../../ApiRequest';
import './Table.css';

/**
 * @function DataTable
 * @description
 * Generic table component that loads and displays data from a database table.
 * @param {Object} props - The component properties.
 * @param {string} props.tableName - The name of the table to fetch from the backend.
 * @param {string} props.title - The table title displayed above the data.
 * @param {string[]} props.columns - An array of column names to display in the table header.
 * @param {string} [props.token] - Optional JWT token for authenticated requests.
 * @returns {JSX.Element} Rendered DataTable component.
 */
export default function DataTable({ tableName, title, columns, token }) {
  /** @member {Array} data - Array holding the fetched table rows. */
  const [data, setData] = useState([]);
  /** @member {boolean} loading - Indicates whether the table is loading. */
  const [loading, setLoading] = useState(true);
  /** @member {string|null} error - Stores any error message if fetch fails. */
  const [error, setError] = useState(null);
  /** @member {boolean} collapsed - Determines whether table is visible. */
  const [collapsed, setCollapsed] = useState(false);

  // Load data on mount
  useEffect(() => {
    api.get('fetch_table', { tableName })
      .then((res) => setData(res))
      .catch((err) => {
        console.error(err);
        setError(`Failed to load ${title.toLowerCase()}`);
      })
      .finally(() => setLoading(false));
  }, [tableName, token, title]);

  if (loading) return <p>Loading {title.toLowerCase()}...</p>;
  if (error) return <p className="error">{error}</p>;

  const total = data.rows?.length || 0;

  return (
    <div className="table-section">
      <div className="table-header">
        <h2>{title} ({total} Rows)</h2>
        <button
          className="collapse-btn"
          onClick={() => setCollapsed((prev) => !prev)}
        >
          {collapsed ? 'Show Table' : 'Hide Table'}
        </button>
      </div>

      {!collapsed && (
        <table className="styled-table">
          <thead>
            <tr>
              {columns.map((col) => (
                <th key={col}>{col}</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {data.rows?.map((row, rowIndex) => (
              <tr key={rowIndex}>
                {columns.map((col) => (
                  <td key={col}>{formatCellValue(row[col])}</td>
                ))}
              </tr>
            ))}
          </tbody>
        </table>
      )}
    </div>
  );
}

/**
 * @function formatCellValue
 * @description
 * Safely formats cell values for display (e.g., date formatting).
 * @param {*} value - The raw value from the database.
 * @returns {string} A human-readable formatted string.
 */
function formatCellValue(value) {
  if (value instanceof Date) {
    return value.toISOString().split('T')[0];
  }
  // handle timestamps (strings that look like dates)
  if (typeof value === 'string' && value.includes('-') && value.includes(':')) {
    return new Date(value).toISOString().split('T')[0];
  }
  return value ?? '';
}
